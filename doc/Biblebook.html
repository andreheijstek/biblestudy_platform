def self.possible_book_names(name)
  biblebook = find_by_full_name(name)
  unless exists?(biblebook)
    biblebook = find_by_abbreviation(name)
    unless exists?(biblebook)
      biblebook = find_names_by_like(name)
    end
  end

  biblebook.map(&:name).uniq
end
