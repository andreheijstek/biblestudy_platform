def populate_bibleverses
  @starting_bibleverse =
    Bibleverse.new({ chapter: starting_chapter_nr, verse: starting_verse_nr })
  @ending_bibleverse =
    Bibleverse.new({ chapter: ending_chapter_nr, verse: ending_verse_nr })
end
